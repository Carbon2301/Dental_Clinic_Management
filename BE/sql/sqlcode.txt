
-- Bảng Patient (Bệnh nhân)
CREATE TABLE patients (
    id INT PRIMARY KEY IDENTITY(1,1),
    full_name VARCHAR(100) NOT NULL,
    dob DATE NOT NULL,
    phone VARCHAR(15),
    sex VARCHAR(10),
    email VARCHAR(100),
	cccd VARCHAR (30) NOT NULL ,
    address VARCHAR(255)
);
GO

ALTER TABLE patients
ADD CONSTRAINT UQ_cccd UNIQUE (cccd);
GO


-- Bảng Appointment (Lịch hẹn)
CREATE TABLE appointments (
    id INT PRIMARY KEY IDENTITY(1,1),
    patient_id INT FOREIGN KEY REFERENCES patients(id),
    appointment_date DATE NOT NULL,
    appointment_time TIME NOT NULL,
    notes VARCHAR(MAX)

);
GO

-- Bảng Employee (Y tá/Bác sĩ)
CREATE TABLE dentists (
    id INT PRIMARY KEY IDENTITY(1,1),
	user_name VARCHAR(50) UNIQUE NOT NULL,
	password VARCHAR(50) NOT NULL,
    full_name VARCHAR(100) NOT NULL,
    position VARCHAR(50),  -- Bác sĩ, Y tá...
    phone VARCHAR(15),
    email VARCHAR(100),
	status INT,
	is_working INT
);
GO


CREATE TABLE roles(
	id INT PRIMARY KEY IDENTITY(1,1),
	role_name VARCHAR(50) NOT NULL
)

CREATE TABLE dentist_role(
	dentist_id INT FOREIGN KEY REFERENCES dentists(id),
	role_id INT FOREIGN KEY REFERENCES roles(id)
)


-- Bảng Treatment (Điều trị)
CREATE TABLE treatments (
    id INT PRIMARY KEY IDENTITY(1,1),
    patient_id INT FOREIGN KEY REFERENCES patients(id),
	dentist_id INT FOREIGN KEY REFERENCES dentists(id),
    treatment_date DATE NOT NULL,
    fees DECIMAL(18, 2) NOT NULL, -- Phí điều trị
    notes VARCHAR(MAX),
	service_name VARCHAR(100) NOT NULL
);
GO
-- Bảng Service (Dịch vụ)
CREATE TABLE services (
    id INT PRIMARY KEY IDENTITY(1,1),
    service_name VARCHAR(100) NOT NULL,
    price DECIMAL(18, 2) NOT NULL
);
GO
CREATE TABLE treatments_services (
    treatment_id INT,
    service_id INT,
    PRIMARY KEY (treatment_id, service_id),
    FOREIGN KEY (treatment_id) REFERENCES treatments(id),
    FOREIGN KEY (service_id) REFERENCES services(id)
);
GO
-- Bảng Prescription (Đơn thuốc) - lưu giá tiền cho từng đơn thuốc
CREATE TABLE prescriptions (
    id INT PRIMARY KEY IDENTITY(1,1),
    treatment_id INT FOREIGN KEY REFERENCES treatments(id),
    medicine_name VARCHAR(100),
	price DECIMAL (18,2) NOT NULL
);
GO
-- Bảng Medicine (Loại thuốc) - Không chứa giá
CREATE TABLE medicines (
    id INT PRIMARY KEY IDENTITY(1,1),
    type NVARCHAR(100) -- Loại thuốc (ví dụ: kháng sinh, giảm đau)
);
GO
CREATE TABLE prescription_medicine (
    prescription_id INT,
    medicine_id INT,
	PRIMARY KEY (prescription_id, medicine_id),
    FOREIGN KEY (prescription_id) REFERENCES prescriptions(id),
    FOREIGN KEY (medicine_id) REFERENCES medicines(id),
    dosage NVARCHAR(50),  -- Liều lượng cho từng thuốc trong đơn
    days INT,  -- Số ngày dùng cho từng thuốc
	price DECIMAL(18, 2) NOT NULL,  -- Giá tiền cho đơn thuốc này
    total_price AS (days * price)  -- Tổng tiền của thuốc dựa trên số ngày dùng
);
GO

-- Bảng Invoice (Hóa đơn)
CREATE TABLE bills (
    id INT PRIMARY KEY IDENTITY(1,1),
    patient_id INT FOREIGN KEY REFERENCES patients(id),
    treatment_id INT FOREIGN KEY REFERENCES treatments(id),
    total_amount DECIMAL(18, 2) NOT NULL,
    date DATE DEFAULT GETDATE()
);
